79      private String getProjectRoot(String project) {
80          String sep = System.getProperty("file.separator");
81          return project.substring(0,project.indexOf("mda"+sep+"src"+sep+"uml"));        
82      }

87      /***
88       * @see java.awt.event.ActionListener#actionPerformed(java.awt.event.ActionEvent)
89       *
90       * Just let the tester know that we got executed.
91       */
92      public void actionPerformed(ActionEvent event) {
93          MavenLauncher launcher = new MavenLauncher();
94          if (projectPath==null)
95              try {
96                  projectPath= AndroMDAModule.getArgoUMLProjectPath();
97              } catch (Exception e1) {
98                  e1.printStackTrace();
99              }
100         if (mavenHome == null)
101             mavenHome = AndroMDAModule.getArgoUMLMavenHome();
102         if (parentFrame == null)
103             parentFrame = AndroMDAModule.getArgoUMLParentFrame();
104         if (dialog==null)
105             buildDialog(parentFrame);
106         if (mavenHome!=null &&(projectPath != null)) {
107             LOG.info("Assume that the model is in $PROJECT/mda/src/uml/");
108             try {
109                 String projectRoot = getProjectRoot(projectPath); 
110                 launcher.setMavenHome(mavenHome);
111                 launcher.setProjectRoot(projectRoot);                
112                 launcher.setStdOut(new TextAreaOutputStream(mavenOutput));          
113                 launcher.setStdErr(new TextAreaOutputStream(mavenError));                
114                 //GUI
115                 dialog.pack();
116                 dialog.toFront();
117                 dialog.setVisible(true);
118                 launcher.start();
119             } catch (Exception e) {
120                 throw new RuntimeException("Problem executing AndroMDA: "+e.getMessage());
121             }
122         } else {
123             throw new RuntimeException("Problem executing AndroMDA: "+ mavenHome +"/"+projectPath);            
124         }
125     }
126 
127     private void buildDialog(Frame owner) {
128         dialog = new ArgoDialog(owner, "Maven output", false);
129         mavenPanel = (JPanel) parent.getSwingEngine().find("mavenPanel");   
130         mavenOutput = (JTextArea) parent.getSwingEngine().find("mavenOutput");
131         mavenError = (JTextArea) parent.getSwingEngine().find("mavenError");
132         dialog.setContent(mavenPanel);  
133     }
