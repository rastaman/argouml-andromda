26  import java.awt.event.WindowAdapter;
27  import java.awt.event.WindowEvent;
28  
29  import javax.swing.JFrame;
30  import javax.swing.JPanel;
31  import javax.swing.JTextArea;
32  
33  /***
34   * A Wizard(test)Frame
35   * @author lmaitre
36   */
37  
38  public class WizardFrame extends JFrame
39  {
40  	
41  	private String descriptor;
42  	
43  	public JTextArea content;
44  	
45  	public WizardFrame(String desc)
46  	{
47  		try{
48              descriptor = desc;
49  			setSize(800,600);
50  			JPanel container = new JPanel();
51  			content = new JTextArea();
52  			content.setColumns(80);
53  			content.setRows(20);
54  			container.add(content);
55  			show();
56  			DefaultWizardDialog wm = new DefaultWizardDialog(this,descriptor);			
57  			wm.show();
58  			this.addWindowListener(new WindowAdapter(){
59  				public void windowClosing(WindowEvent we)
60  				{
61  					System.exit(0);
62  				}
63  			});
64  		}catch(Exception ex){
65  			ex.printStackTrace();
66  		}
67  	}
68  
69  	public static void main(String args[])
70  	{
71  		new WizardFrame(args[0]);
72  	}
73  }
