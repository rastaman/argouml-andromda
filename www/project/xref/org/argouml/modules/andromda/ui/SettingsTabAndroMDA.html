143     private void refreshModulesPath() {
144         LOG.info("Assume that the profiles are in $ANDROMDA_HOME/andromda/xml.zips");        
145         String sep = System.getProperty("file.separator");
146         String andromdaRepo = andromdaHome.getText() + sep + "andromda" + sep + "xml.zips";
147         String path = System.getProperty("org.argouml.model.modules_search_path");
148         if (path!=null) {
149             boolean alreadyIn = false;
150             String[] paths = path.split(",");
151             for (int i=0;i<paths.length;i++) {
152                 if (paths[i].equals(andromdaRepo))
153                     alreadyIn=true;
154             }
155             if (!alreadyIn) {
156                 if (paths.length>0)
157                     path += "," + andromdaRepo;
158                 else
159                     path = andromdaRepo;
160             }
161         } else
162             path = andromdaRepo;
163         System.setProperty("org.argouml.model.modules_search_path",path);
164         
165     }

182         String keyAndromdaHome = Configuration.getString(SettingsTabAndroMDA.KEY_ANDROMDA_HOME);

184             andromdaHome.setText(keyAndromdaHome);
185         refreshModulesPath();
186     }
